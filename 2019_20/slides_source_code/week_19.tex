\frametitle{Grievances vs feasibility}
Grievances: \pause People \emph{want} to rebel \pause
\begin{itemize}
\item Socio-economic inequalities \pause
	\begin{itemize}
	\item[--] Income \pause
	\item[--] Access to resources \pause
	\item[--] Public goods \pause
	\end{itemize}
\item Discrimination \pause
	\begin{itemize}
	\item[--] Ethnic fractionalization \pause
	\item[--] Polarization \pause
	\item[--] Segregation
	\end{itemize} \pause
\item Repressive government
\end{itemize}
}

\frame{
\frametitle{Grievances vs feasibility}
Feasibility: \pause People are \emph{able} to rebel \pause
\begin{itemize}
\item Funding sources \pause
	\begin{itemize}
	\item[--] Natural resources \pause
	\item[--] External support \pause
	\end{itemize}
\item Low cost for recruitment and fighting \pause
	\begin{itemize}
	\item[--] Weak states \pause
	\item[--] Poverty 
	\end{itemize}
\end{itemize}
}

\frame{
\frametitle{What makes conflict feasible?}
\textbf{Feasibility}: easier to recruit, finance, and sustain rebellion \pause

\textbf{Greed/Opportunity}: make conflict ``attractive'' and/or a viable option \pause
\begin{itemize}
\item Presence of lootable resources (diamonds, heroin, cocaine etc.) \pause \textcolor{red}{anything else?} \pause
\item Mountainous terrain \pause
\item Weak states \pause
\item Poverty \pause
\item Youth bulges (unemployed young males) \pause
\item Availability of arms (surrounding conflicts!)
\end{itemize}
}

\frame{
\frametitle{Feasibility Hypothesis}
\textbf{Collier and Hoeffler (2004):} \pause
\begin{itemize}
\item Where rebellion is feasible it will occur \pause
\item Motivation/Grievances are more relevant for propaganda purposes \pause
\item Rebel army is really expensive \pause
\item \textbf{Financial and military viability $=$ Feasibility}. \pause How can we measure it?
	\begin{itemize}
	\item[--] Low GDP/capita \pause 
	\item[--] Growth of GDP/capita \pause
	\item[--] Primary commodity exports \pause
	\item[--] Proportion of young men \pause 
	\item[--] French colony \pause
	\item[--] Mountainous terrain
	\end{itemize}
\end{itemize}
}

\frame{
\frametitle{Discussion}
\centering
\textcolor{red}{What type of cause for a civil conflict is feasibility (greed)?} \pause \textcolor{red}{Is it a necessary cause?} \pause \textcolor{red}{Is it a sufficient cause?} \pause \textcolor{red}{What about grievances?}
}

\frame{
\frametitle{Criticism to greed \emph{and} grievances}
\textbf{Wood (2003)}: Collective Action Problem in El Salvador
\begin{itemize}
\item Moral and emotional imperative vs. material gains: \\ \pause
Material grievances motivate rebellion\ldots \pause \textbf{BUT} emotional and moral reasons were essential \pause
	\begin{itemize}
	\item[--] The government as a profoundly \emph{unjust} authority
	\item[--] \emph{Pride, dignity, and joy} with participating in the insurgency
	\end{itemize} \pause
\item \textbf{Horizontal Inequalities} \pause
	\begin{itemize}
	\item[--] \emph{What kind} of grievances matter? When and how? \pause
	\item[--] Economic, political, social inequality \emph{between groups}
	\end{itemize}
\end{itemize}
}
